import os

# Define the base directory for the dataset
base_dir = 'C:/Users/HP/Downloads/nuaa_dataset/raw'

def load_image_paths_and_labels(base_dir, subdirs, label):
    paths = []
    for subdir in subdirs:
        dir_path = os.path.join(base_dir, subdir)
        for root, dirs, files in os.walk(dir_path):
            for file in files:
                if file.lower().endswith(('.png', '.jpg', '.jpeg')):
                    full_path = os.path.join(root, file)
                    paths.append((full_path, label))
    return paths

# Define the subdirectories and labels

from tensorflow.keras.applications import MobileNetV2
from tensorflow.keras import layers, models
from tensorflow.keras import regularizers
def build_liveness_model():
    base_model = MobileNetV2(weights='imagenet', include_top=False, input_shape=(224,224,3))
    
    # Fine-tune some layers by setting them trainable
    for layer in base_model.layers[-30:]:  # Fine-tuning the last 30 layers
        layer.trainable = True
    model = models.Sequential([
        base_model,
        layers.GlobalAveragePooling2D(),
        layers.Dense(512, activation='relu', kernel_regularizer=regularizers.l2(0.01)),
        layers.Dropout(0.5),
        layers.BatchNormalization(),
        layers.Dense(256, activation='relu', kernel_regularizer=regularizers.l2(0.01)),
        layers.Dropout(0.5),
        layers.BatchNormalization(),
        layers.Dense(1, activation='sigmoid')
        
    ])

    

    model.compile(optimizer='adam', loss='binary_crossentropy', metrics=['accuracy'])
    return model

# Build and summarize the new model
model = build_liveness_model()
model.summary()

import cv2
import numpy as np

def preprocess_frame(frame):
    # Resize and normalize the frame for model prediction
    frame_resized = cv2.resize(frame, (160,160))
    frame_normalized = frame_resized / 255.0
    frame_expanded = np.expand_dims(frame_normalized, axis=0)
    return frame_expanded

def predict_liveness(frame, model):
    preprocessed_frame = preprocess_frame(frame)
    prediction = model.predict(preprocessed_frame)
    return prediction[0][0]  # Assuming a binary classification (0 or 1)

def real_time_detection(model):
    # Initialize the video capture
    cap = cv2.VideoCapture(0)  # 0 for the default camera

    if not cap.isOpened():
        print("Error: Could not open video capture.")
        return

    print("Press 'q' to quit.")
    
    while True:
        # Capture frame-by-frame
        ret, frame = cap.read()
        if not ret:
            print("Error: Failed to capture image.")
            break
        
        # Make a prediction
        liveness_score = predict_liveness(frame, model)
        print("score is ",liveness_score)
        # Define a threshold for liveness detection
        threshold = 0.5  # Adjust based on your model and evaluation
        label = 'Real' if liveness_score > threshold else 'Fake'
        
        # Draw the result on the frame
        cv2.putText(frame, f'Liveness: {label}', (10, 30), cv2.FONT_HERSHEY_SIMPLEX, 1, (0, 255, 0) if label == 'Real' else (0, 0, 255), 2, cv2.LINE_AA)
        cv2.imshow('Liveness Detection', frame)
        print(label)
        # Break the loop on 'q' key press
        key = cv2.waitKey(1) & 0xFF
        if key == ord('q'):
            break

    # Release the capture and close windows
    cap.release()
    cv2.destroyAllWindows()

# Run the real-time detection
real_time_detection(model)
